Local Codebase Implementations

Local helpers referenced by the tests above. Each helper appears once even when called from multiple cases.

openRecipeCard(page, slug)

export async function openRecipeCard(page: Page, slug: string) {
  await page.goto(`/recipes/${slug}`)
  await page.waitForSelector('[data-recipe-loaded="true"]')
}

setServings(page, value)

export async function setServings(page: Page, value: number) {
  await page.getByTestId('servings-input').fill(String(value))
  await page.getByTestId('servings-input').press('Enter')
  await page.waitForFunction(
    (v) => document.querySelector('[data-testid="servings-value"]')?.textContent === String(v),
    value,
  )
}
